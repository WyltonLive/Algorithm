# **回溯算法**

## 理论基础

- 因为回溯的本质是穷举，穷举所有可能，然后选出我们想要的答案，如果想让回溯法高效一些，可以加一些剪枝的操作，但也改不了回溯法就是穷举的本质。
- **for循环可以理解是横向遍历，backtracking（递归）就是纵向遍历**
- 模板：

``` C++
void backtracking(参数) {
    if (终止条件) {
        存放结果;
        return;
    }

    for (选择：本层集合中元素（树中节点孩子的数量就是集合的大小）) {
        处理节点;
        backtracking(路径，选择列表); // 递归
        回溯，撤销处理结果
    }
}
```

class Solution {
private:
    vector<vector<int>> result; // 存放符合条件结果的集合
    vector<int> path; // 用来存放符合条件结果
    void backtracking(int n, int k, int start) {
        if (path.size() == k) {
            result.push_back(path);
            return;
        }
        //n - i >= k - path.size()
        //由于是左闭区间，故要加一
        for (int i = start; i <= n - (k - path.size()) + 1; i++) {
            path.push_back(i); // 处理节点
            backtracking(n, k, i + 1); // 递归
            path.pop_back(); // 回溯，撤销处理的节点
        }
    }
public:
    vector<vector<int>> combine(int n, int k) {
        result.clear(); // 可以不写
        path.clear();   // 可以不写
        backtracking(n, k, 1);
        return result;
    }
};

### **每个数字不可重复取用，所给数组元素有重复，但结果不可重复**

![alt text](./image-2.png)

- **每个数字不可重复取用**：递归函数参数 **i + 1**
- **元素有重复**：由于还要去重，所以得先排序，同一树枝上元素有重复没有关系，因为**同一结果数组里面元素有重复符合题意，不符合的是整个数组重复**
- **结果不可重复** ：同一树层上的不可重复， 因为这意味着**接下来递归的 内容 和上一个重复的元素一样**（数组已经排序，且递归是按下标递归）

![alt text](./image-3.png)

---

- used[i - 1] == true，说明同一树枝candidates[i - 1]使用过，因为**每次放入会复制为true，而同树层横向跳跃前，不会经过used[i]=false这一操作**
- used[i - 1] == false，说明同一树层candidates[i - 1]使用过，因为**从该节点遍历到该树枝完后才跳到同一个树层的下一个节点，而跳出时会撤销：used[i]=false**
- candidates[i] = candidates[i - 1] 条件一加，表明当前元素和上一层的相同
- 合在一起，表示该元素在横向或竖向上重复了

---

```cpp
class Solution {
private:
    vector<vector<int>> result;
    vector<int> path;
    void backTracking(vector<int>& candidates, int target, int startIndex, 
    int sum, vector<bool> &used){
        if(sum > target){
            return;
        }else if(sum == target){
            result.push_back(path);
            return;
        }
        for(int i = startIndex; i < candidates.size() && 
        sum + candidates[i] <= target; i++){
            //同一树层的还相同的元素去掉，因为这代表后面的回溯会重复
            // used[i - 1] == true，说明同一树枝candidates[i - 1]使用过
            // used[i - 1] == false，说明同一树层candidates[i - 1]使用过
            //candidates[i] = candidates[i - 1] 条件一加，表明当前元素和上一层的相同
            if(i > 0 && candidates[i] == candidates[i - 1] && used[i - 1] == false){
                continue;
            }
            path.push_back(candidates[i]);
            sum += candidates[i];
            used[i] = true;
            backTracking(candidates, target, i + 1, sum, used);
            used[i] = false;//记得撤销used数组
            sum -= candidates[i];
            path.pop_back();
        }
    }

public:
    vector<vector<int>> combinationSum2(vector<int>& candidates, int target) {
        //这里不能放在private里，因为那里candidates还没有定义
        vector<bool> used(candidates.size(), false);
        //排序后相同的元素相邻，candidates[i] == candidates[i-1] 才能检测出所有重复元素
        sort(candidates.begin(), candidates.end());
        backTracking(candidates, target, 0, 0, used);
        return result;
    }
};
```

class Solution {
private:
    vector<vector<string>> result;
    vector<string> path;
    bool isPalindrome(const string &s, int start, int end){
        for(int i = start, j = end; i < j; i ++, j --){
            if(s[i] != s[j]){
                return false;
            }
        }
        return true;
    }

    void backTracking(const string &s, int startIndex){
        //由于题目没有给取几个数，则递归深度为字符串长度
        //（因为要每一个字符都分割），故直接用 startIndex >= s.size() 为终止条件
        if(startIndex >= s.size()){
            //这里可以直接放
            //因为不符合的在上一层已经被continue掉了
            result.push_back(path);
            return;
        }
        for(int i = startIndex; i < s.size(); i ++){
            if(isPalindrome(s, startIndex, i)){
                //从startIndex开始截取， 每一种可能的长度都试
                string str = s.substr(startIndex, i - startIndex + 1);
                path.push_back(str);
            }else{
                continue;
            }
            backTracking(s, i + 1);//是i + 1， 不是startIndex + 1
            path.pop_back();
        }
    }

public:
    vector<vector<string>> partition(string s) {
        backTracking(s, 0);
        return result;
    }
};

![alt text](./image-6.png)

![alt text](./image-7.png)

```cpp
class Solution {
private:
    vector<string> result;
    bool isValid(const string &s, int start, int end){
        if(start > end){
            return false;
        }
        if(s[start] == '0' && start != end){
            return false;
        }
        //以下是整数分解
        int num = 0;
        for(int i = start; i <= end; i ++){
            if(s[i] > '9' || s[i] < '0'){
                return false;
            }
            num = num * 10 + (s[i] - '0');
            if(num > 255){
                return false;
            }
        }
        return true;
    }

    void backTracking(string &s, int startIndex, int pointNum){
        if(pointNum == 3){
            //到了第四个也即最后一个，后面不需要加'.'
            //如果在让函数进入下面的for循环，会多加'.'，且让pointNum继续增加
            //所以这里一定要再加一个if语句，判断最后一个数是否符合
            if(isValid(s, startIndex, s.size() - 1)){
                result.push_back(s);
                return;
            }
        }
        for(int i = startIndex; i < s.size(); i ++){
            if(isValid(s, startIndex, i)) {
                s.insert(s.begin() + i + 1, '.');
                pointNum ++;
                //注意是i + 2,因为有'.'
                backTracking(s, i + 2, pointNum);
                pointNum --;
                //注意是 i + 1，不是startIndex + 1
                s.erase(s.begin() + i + 1);
            }else{
                //不用continue, 因为如果从startIndex到i不满足，那到i + 1必然不满足
                break;
            }
        }
    }

public:
    vector<string> restoreIpAddresses(string s) {
        result.clear();
        if(s.size() <  4 || s.size() > 12){
            return result;
        }
        backTracking(s, 0, 0);
        return result;
    }
};

  - 排列：**无start，全数组遍历**，即使跳过当前这个相同元素，后续还能在其他分支中选取其他索引的元素，不会影响有效排列的生成（只是多走一些分支，最终结果完整
  - 组合：**有start**，组合的start=i+1逻辑是 “单向推进，不回头”，一旦在当前树枝上因为used[i-1] == true跳过某个元素，后续没有任何机会再选取该元素来组成有效组合
- 性能：**对于排列问题，树层上去重和树枝上去重，都是可以的，但是树层上去重效率更高！**

class Solution {
private:
// unordered_map<出发机场, map<到达机场, 航班次数>> targets
unordered_map<string, map<string, int>> targets;
bool backtracking(int ticketNum, vector<string>& result) {
    if (result.size() == ticketNum + 1) {
        return true;
    }
    for (pair<const string, int>& target : targets[result[result.size() - 1]]) {
        if (target.second > 0 ) { // 记录到达机场是否飞过了
            result.push_back(target.first);
            target.second--;
            if (backtracking(ticketNum, result)) return true;
            result.pop_back();
            target.second++;
        }
    }
    return false;
}
public:
    vector<string> findItinerary(vector<vector<string>>& tickets) {
        targets.clear();
        vector<string> result;
        for (const vector<string>& vec : tickets) {
            targets[vec[0]][vec[1]]++; // 记录映射关系
        }
        result.push_back("JFK"); // 起始机场
        backtracking(tickets.size(), result);
        return result;
    }
};

class Solution {
private:
vector<vector<string>> result;
// n 为输入的棋盘大小
// row 是当前递归到棋盘的第几行了
void backtracking(int n, int row, vector<string>& chessboard) {
    if (row == n) {
        result.push_back(chessboard);
        return;
    }
    for (int col = 0; col < n; col++) {
        if (isValid(row, col, chessboard, n)) { // 验证合法就可以放
            chessboard[row][col] = 'Q'; // 放置皇后
            backtracking(n, row + 1, chessboard);
            chessboard[row][col] = '.'; // 回溯，撤销皇后
        }
    }
}
bool isValid(int row, int col, vector<string>& chessboard, int n) {
    // 检查列
    for (int i = 0; i < row; i++) { // 这是一个剪枝
        if (chessboard[i][col] == 'Q') {
            return false;
        }
    }
    // 检查 45度角是否有皇后
    for (int i = row - 1, j = col - 1; i >=0 && j >= 0; i--, j--) {
        if (chessboard[i][j] == 'Q') {
            return false;
        }
    }
    // 检查 135度角是否有皇后
    for(int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++) {
        if (chessboard[i][j] == 'Q') {
            return false;
        }
    }
    return true;
}
public:
    vector<vector<string>> solveNQueens(int n) {
        result.clear();
        std::vector<std::string> chessboard(n, std::string(n, '.'));
        backtracking(n, 0, chessboard);
        return result;
    }
};
```

class Solution {
private:
bool backtracking(vector<vector<char>>& board) {
    for (int i = 0; i < board.size(); i++) {        // 遍历行
        for (int j = 0; j < board[0].size(); j++) { // 遍历列
            if (board[i][j] == '.') {
                for (char k = '1'; k <= '9'; k++) {     // (i, j) 这个位置放k是否合适
                    if (isValid(i, j, k, board)) {
                        board[i][j] = k;                // 放置k
                        if (backtracking(board)) return true; // 如果找到合适一组立刻返回
                        board[i][j] = '.';              // 回溯，撤销k
                    }
                }
                return false;  // 9个数都试完了，都不行，那么就返回false
            }
        }
    }
    return true; // 遍历完没有返回false，说明找到了合适棋盘位置了
}
bool isValid(int row, int col, char val, vector<vector<char>>& board) {
    for (int i = 0; i < 9; i++) { // 判断行里是否重复
        if (board[row][i] == val) {
            return false;
        }
    }
    for (int j = 0; j < 9; j++) { // 判断列里是否重复
        if (board[j][col] == val) {
            return false;
        }
    }
    int startRow = (row / 3) * 3;
    int startCol = (col / 3) * 3;
    for (int i = startRow; i < startRow + 3; i++) { // 判断9方格里是否重复
        for (int j = startCol; j < startCol + 3; j++) {
            if (board[i][j] == val ) {
                return false;
            }
        }
    }
    return true;
}
public:
    void solveSudoku(vector<vector<char>>& board) {
        backtracking(board);
    }
};
